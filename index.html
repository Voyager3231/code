# Linear Search Function
def linear_search(arr, key):
    for i in range(len(arr)):
        if arr[i] == key:
            return i
    return -1

# Binary Search Function
def binary_search(arr, key):
    arr.sort()  # Binary search requires a sorted list
    low, high = 0, len(arr) - 1
    while low <= high:
        mid = (low + high) // 2
        if arr[mid] == key:
            return mid
        elif arr[mid] < key:
            low = mid + 1
        else:
            high = mid - 1
    return -1

def selection_sort(arr):
    for i in range(len(arr)):
        min_idx = i
        for j in range(i + 1, len(arr)):
            if arr[j] < arr[min_idx]:
                min_idx = j
        arr[i], arr[min_idx] = arr[min_idx], arr[i]
    return arr


def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
    return arr

class TextEditor:
    def __init__(self):
        self.undo_stack = []
        self.redo_stack = []
        self.text = ""

    def make_change(self, change):
        self.undo_stack.append(self.text)
        self.text += change
        self.redo_stack.clear()

    def undo(self):
        if self.undo_stack:
            self.redo_stack.append(self.text)
            self.text = self.undo_stack.pop()
        else:
            print("Nothing to undo!")

    def redo(self):
        if self.redo_stack:
            self.undo_stack.append(self.text)
            self.text = self.redo_stack.pop()
        else:
            print("Nothing to redo!")

    def display(self):
        print("\nCurrent Document:", self.text)

class StudentLinkedList:
    def __init__(self):
        self.head = None

    def add_student(self, roll_no, name, marks):
        new_node = StudentNode(roll_no, name, marks)
        if self.head is None:
            self.head = new_node
        else:
            current = self.head
            while current.next:
                current = current.next
            current.next = new_node
        print(" Student record added.")

    def delete_student(self, roll_no):
        current = self.head
        prev = None
        while current:
            if current.roll_no == roll_no:
                if prev:
                    prev.next = current.next
                else:
                    self.head = current.next
                print(" Student record deleted.")
                return
            prev = current
            current = current.next
        print(" Student not found.")

    def update_student(self, roll_no, new_name, new_marks):
        current = self.head
        while current:
            if current.roll_no == roll_no:
                current.name = new_name
                current.marks = new_marks
                print(" Student record updated.")
                return
            current = current.next
        print(" Student not found.")

    def search_student(self, roll_no):
        current = self.head
        while current:
            if current.roll_no == roll_no:
                print(f" Found: Roll No: {current.roll_no}, Name: {current.name}, Marks: {current.marks}")
                return
            current = current.next
        print(" Student not found.")

            print(" No records to display.")
            return

        print("\n Student Records:")
        for s in students:
            print(f"Roll No: {s[0]}, Name: {s[1]}, Marks: {s[2]}")

class HashTable:
    def __init__(self, size=10):
        self.size = size
        self.table = [[] for _ in range(size)]

    def hash_function(self, key):
        return key % self.size

    def insert(self, key, value):
        index = self.hash_function(key)
        for pair in self.table[index]:
            if pair[0] == key:
                pair[1] = value
                return
        self.table[index].append([key, value])

    def search(self, key):
        index = self.hash_function(key)
        for pair in self.table[index]:
            if pair[0] == key:
                return pair[1]
        return None

    def delete(self, key):
        index = self.hash_function(key)
        self.table[index] = [pair for pair in self.table[index] if pair[0] != key]

    def display(self):
        print("\nüì¶ Hash Table Contents:")
        for i, bucket in enumerate(self.table):
            print(f"Index {i}: {bucket}")

class LinearProbingHashTable:
    def __init__(self, size=10):
        self.size = size
        self.table = [None] * size

    def hash_function(self, key):
        return key % self.size

    def insert(self, key):
        index = self.hash_function(key)
        start_index = index  # to detect full table loop
        while self.table[index] is not None:
            index = (index + 1) % self.size
            if index == start_index:
                print("‚ùå Hash table is full! Cannot insert.")
                return
        self.table[index] = key
        print(f"‚úÖ Key {key} inserted at index {index}.")

    def search(self, key):
        index = self.hash_function(key)
        for _ in range(self.size):
            if self.table[index] == key:
                return index
            index = (index + 1) % self.size
        return -1

    def delete(self, key):
        index = self.search(key)
        if index != -1:
            self.table[index] = None
            print(f"üóëÔ∏è Key {key} deleted from index {index}.")
        else:
            print("‚ùå Key not found!")

    def display(self):
        print("\nüì¶ Hash Table:")

class Bucket:
    def __init__(self, depth, size):
        self.depth = depth
        self.size = size
        self.items = {}

    def is_full(self):
        return len(self.items) >= self.size


class ExtendibleHashTable:
    def __init__(self, bucket_size=2):
        self.global_depth = 1
        self.bucket_size = bucket_size
        self.directory = [Bucket(1, bucket_size) for _ in range(2)]

    def _hash(self, key):
        return key & ((1 << self.global_depth) - 1)

    def insert(self, key, value):
        i = self._hash(key)
        b = self.directory[i]

        if key in b.items or not b.is_full():
            b.items[key] = value
            print(f"{key}:{value} inserted in bucket {i}")
            return

        print(f"Bucket {i} full, splitting...")
        self.split(i)
        self.insert(key, value)

    def split(self, i):
        old = self.directory[i]
        old.depth += 1
        if old.depth > self.global_depth:
            self.directory *= 2
            self.global_depth += 1

        new = Bucket(old.depth, self.bucket_size)
        for j in range(len(self.directory)):
            if self.directory[j] is old and (j >> (old.depth - 1)) & 1:
                self.directory[j] = new

        data = list(old.items.items())
        old.items.clear()
        for k, v in data:
            self.insert(k, v)

    def search(self, key):
        i = self._hash(key)
        b = self.directory[i]
        if key in b.items:
            print(f"{key} found: {b.items[key]}")
        else:
            print(f"{key} not found")

    def delete(self, key):
        i = self._hash(key)
        b = self.directory[i]
        if key in b.items:
            del b.items[key]
            print(f"{key} deleted")
        else:
            print(f"{key} not found")

    def display(self):
        print("\nDirectory:")
        seen = set()
        for i, b in enumerate(self.directory):
            if id(b) not in seen:
                seen.add(id(b))
                print(f"Bucket {i} (depth={b.depth}): {b.items}")

def dfs(start, visited=None):
    if visited is None:
        visited = set()
    visited.add(start)
    print(start, end=" ")
    for node in graph[start]:
        if node not in visited:
            dfs(node, visited)

# BFS Function
# -------------------
def bfs(start):
    visited = set()
    queue = deque([start])
    while queue:
        node = queue.popleft()
        if node not in visited:
            print(node, end=" ")
            visited.add(node)
            for n in graph[node]:
                if n not in visited:
                    queue.append(n)

class BST:
    def __init__(self):
        self.root = None

    def insert(self, key):
        self.root = self._insert(self.root, key)

    def _insert(self, root, key):
        if not root:
            return Node(key)
        if key < root.key:
            root.left = self._insert(root.left, key)
        elif key > root.key:
            root.right = self._insert(root.right, key)
        return root

    def search(self, key):
        root = self.root
        while root:
            if key == root.key:
                return True
            elif key < root.key:
                root = root.left
            else:
                root = root.right
        return False

    def delete(self, key):
        self.root = self._delete(self.root, key)

    def _delete(self, root, key):
        if not root:
            return root
        if key < root.key:
            root.left = self._delete(root.left, key)
        elif key > root.key:
            root.right = self._delete(root.right, key)
        else:
            # Node found
            if not root.left:
                return root.right
            elif not root.right:
                return root.left
            # Node with two children
            temp = self._min_value(root.right)
            root.key = temp.key
            root.right = self._delete(root.right, temp.key)
        return root

    def _min_value(self, node):
        current = node
        while current.left:
            current = current.left
        return current

    def inorder(self, root):
        if root:
            self.inorder(root.left)
            print(root.key, end=' ')
            self.inorder(root.right)

class CityBST:
    def __init__(self):
        self.root = None

    def insert(self, city, population):
        self.root = self._insert_recursive(self.root, city, population)

    def _insert_recursive(self, root, city, population):
        if root is None:
            return Node(city, population)
        if city.lower() < root.city.lower():
            root.left = self._insert_recursive(root.left, city, population)
        elif city.lower() > root.city.lower():
            root.right = self._insert_recursive(root.right, city, population)
        return root

    def search(self, city):
        comparisons = 0
        root = self.root
        while root:
            comparisons += 1
            if city.lower() == root.city.lower():
                print(f"‚úÖ Found {city} with population {root.population} (Comparisons: {comparisons})")
                return root
            elif city.lower() < root.city.lower():
                root = root.left
            else:
                root = root.right
        print(f"‚ùå {city} not found after {comparisons} comparisons.")
        return None

    def update(self, city, new_population):
        node = self.search(city)
        if node:
            node.population = new_population

        self.root = self._delete_recursive(self.root, city)
